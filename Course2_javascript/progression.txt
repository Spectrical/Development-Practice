Part 62: PROFILE LOOKUP
var contacts = [
  {
    "firstName": "Jeremy",
    "lastName": "Doughs",
    "contactNumber": 08382182,
    "likes": [
      "eating pizza", "programming"
    ]
  },
  {
    "firstName": "James",
    "lastName": "Triot",
    "contactNumber": 08736829,
    "likes": [
      "working at a pizza factory", "jogging"
    ]
  }
]

function lookupProfile(name, prop) {
  for (var i = 0; i < contacts.length; i++)
  {
    if(contacts[i].firstName === name) {
      return contacts[i][prop] || "Can't find the property you are looking for."
    }
  } return "No contacts found.";
}
console.log(lookupProfile("James", "likes"))

Part 76: Use the Rest Operator with Function Parameters. INTRODUCING "reduce()"
const sum = (function() {
  return function sum(x, y, z, a, b, c) {
    const args = [x, y, z, a, b, c];
    return args.reduce((a, b) => a + b, 0);
  };
})();
console.log(sum(1, 2, 3, 4, 5, 6));

// INTRODUCTION OF "..." aka "Rest" OPERATOR

const sum = (function() {
  return function sum(...args) {
    return args.reduce((a, b) => a + b, 0);
  };
})();
console.log(sum(1, 2, 3, 4, 5, 6));
